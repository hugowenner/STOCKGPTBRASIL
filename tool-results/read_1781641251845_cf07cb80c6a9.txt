    87→function formatNumber(n: number | null | undefined, decimals = 2): string {
    88→  if (n === null || n === undefined) return 'N/A';
    89→  if (Math.abs(n) >= 1e12) return `${(n / 1e12).toFixed(1)}T`;
    90→  if (Math.abs(n) >= 1e9) return `${(n / 1e9).toFixed(1)}B`;
    91→  if (Math.abs(n) >= 1e6) return `${(n / 1e6).toFixed(1)}M`;
    92→  if (Math.abs(n) >= 1e3) return `${(n / 1e3).toFixed(1)}K`;
    93→  return n.toFixed(decimals);
    94→}
    95→
    96→function formatPercent(n: number | null | undefined): string {
    97→  if (n === null || n === undefined) return 'N/A';
    98→  return `${(n * 100).toFixed(1)}%`;
    99→}

   101→function formatPrice(n: number | null | undefined, currency = 'R$'): string {
   102→  if (n === null || n === undefined) return 'N/A';
   103→  return `${currency} ${n.toFixed(2)}`;
   104→}

   106→function formatDate(d: string | null | undefined): string {
   107→  if (!d) return 'N/A';
   108→  return new Date(d).toLocaleDateString('pt-BR');

   111→function getScoreColor(score: number): string {
   112→  if (score >= 70) return COLORS.success;
   113→  if (score >= 50) return COLORS.accent;
   114→  if (score >= 30) return '#f97316';
   115→  return COLORS.danger;
   116→}

   118→/**
   119→ * Safely parse a value that might be a JSON string or an array.
   120→ * The API sometimes returns array fields as JSON-encoded strings.
   121→ */
   122→function safeArray(val: unknown): string[] {
   123→  if (Array.isArray(val)) return val;
   124→  if (typeof val === 'string') {
   125→    try {
   126→      const parsed = JSON.parse(val);
   127→      return Array.isArray(parsed) ? parsed : [val];
   128→    } catch {
   129→      return [val];
   130→    }
   131→  }
   132→  return [];
   133→}

   144→// Score Badge
   145→// ============================================================
   146→function ScoreBadge({ score, grade, size = 'md' }: { score: number; grade?: string; size?: 'sm' | 'md' | 'lg' }) {
   147→  const color = getScoreColor(score);
   148→  const sizes = {
   149→    sm: 'w-10 h-10 text-xs',
   150→    md: 'w-14 h-14 text-sm',
   151→    lg: 'w-20 h-20 text-lg',
   152→  };
   153→
   154→  return (
   155→    <div className="flex flex-col items-center gap-1">
   156→      <div
   157→        className={`${sizes[size]} rounded-full flex items-center justify-center font-bold text-white`}
   158→        style={{ backgroundColor: color }}
   159→      >
   160→        {score.toFixed(0)}
   161→      </div>
   162→      {grade && (
   163→        <span className="text-xs font-semibold" style={{ color: GRADE_COLORS[grade] || COLORS.muted }}>
   164→          {grade}
   165→        </span>
   166→      )}
   167→    </div>
   168→  );
   169→}

   606→function UpdateButton({ label, icon, endpoint }: { label: string; icon: React.ReactNode; endpoint: string }) {
   607→  const [loading, setLoading] = useState(false);
   608→  const [result, setResult] = useState<string | null>(null);
   609→
   610→  const handleClick = async () => {
   611→    setLoading(true);
   612→    setResult(null);
   613→    try {
   614→      const apiMethod = endpoint as keyof typeof api;
   615→      const res = await (api[apiMethod] as () => Promise<any>)();
   616→      setResult(`OK: ${res.stocksProcessed} ações processadas`);
   617→    } catch (e) {
   618→      setResult('Erro na atualização');
   619→    }
   620→    setLoading(false);
   621→    setTimeout(() => setResult(null), 5000);
   622→  };
   623→
   624→  return (
   625→    <button
   626→      onClick={handleClick}
   627→      disabled={loading}
   628→      className="flex items-center gap-2 px-4 py-2.5 rounded-xl bg-slate-800/50 border border-slate-700/50 text-sm font-medium text-slate-300 hover:bg-slate-700/50 hover:text-white transition-all disabled:opacity-50"
   629→    >
   630→      {loading ? <Loader2 className="w-4 h-4 animate-spin" /> : icon}
   631→      {label}
   632→      {result && <span className="text-xs text-emerald-400 ml-2">{result}</span>}
   633→    </button>
   634→  );
   635→}
